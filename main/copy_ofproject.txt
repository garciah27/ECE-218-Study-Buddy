#include "freertos/FreeRTOS.h"
#include "driver/gpio.h"
#include "esp_adc/adc_oneshot.h"
#include "driver/ledc.h"
#include <freertos/task.h>
#include <hd44780.h>
#include <esp_idf_lib_helpers.h>
#include <stdio.h>

// LEDs
#define GREEN_LED GPIO_NUM_13      // Ignition enabled
#define RED_LED  GPIO_NUM_14      // Engine running

// Buttons
#define DRI_SEAT GPIO_NUM_4
#define PASS_SEAT GPIO_NUM_5
#define DRI_BELT GPIO_NUM_6
#define PASS_BELT GPIO_NUM_7
#define IGNI_BUTTON GPIO_NUM_10

// Buzzer
#define BUZZER GPIO_NUM_12

// Potentiometers
#define MODE_POT ADC_CHANNEL_2
#define DELAY_POT ADC_CHANNEL_9

// ADC config
#define ADC_ATTEN ADC_ATTEN_DB_12
#define BITWIDTH ADC_BITWIDTH_12

// Servo PWM
#define LEDC_TIMER LEDC_TIMER_0
#define LEDC_MODE LEDC_LOW_SPEED_MODE
#define LEDC_CHANNEL LEDC_CHANNEL_0
#define LEDC_OUTPUT_IO 19
#define LEDC_DUTY_RES LEDC_TIMER_13_BIT
#define LEDC_FREQUENCY 50

// Wiper angles
#define WIPERS_MIN 245
#define WIPERS_MAX 620

// Thresholds
#define OFF_THRESH 3170
#define INT_THRESH 2377
#define LOW_THRESH 1585
#define HIGH_THRESH 793

#define SHORT_THRESH 1057
#define MED_THRESH 2113

#define LOOP_DELAY_MS 25
#define BUZZER_TIME 500

// Delay times
int short_del_ms = 1000;
int med_del_ms = 3000;
int long_del_ms = 5000;

// Wiper speed steps
int low_steps = 65;
int high_steps = 30;

//variables
volatile bool ignition = false;
volatile bool wipe_off = true;
volatile bool wipe_int = false;
volatile bool wipe_low = false;
volatile bool wipe_high = false;

volatile bool short_del = true;
volatile bool med_del = false;
volatile bool long_del = false;

//
void perform_wipe_cycle(int steps);
void wiper_control_task(void *pvParameters);
static void ledc_init(void);
static void buttons_init(void);
static void leds_init(void);
static void buzzer_init(void);


void app_main(void)
{
    // ADC
    adc_oneshot_unit_init_cfg_t init_config = {
        .unit_id = ADC_UNIT_1,
    };
    adc_oneshot_unit_handle_t adc_handle;
    ESP_ERROR_CHECK(adc_oneshot_new_unit(&init_config, &adc_handle));
    adc_oneshot_chan_cfg_t config = {
        .atten = ADC_ATTEN,
        .bitwidth = BITWIDTH,
    };
    ESP_ERROR_CHECK(adc_oneshot_config_channel(adc_handle, MODE_POT, &config));
    ESP_ERROR_CHECK(adc_oneshot_config_channel(adc_handle, DELAY_POT, &config));

    ledc_init();
    leds_init();
    buttons_init();
    buzzer_init();

    // LCD init
    hd44780_t lcd = {
        .write_cb = NULL,
        .font = HD44780_FONT_5X8,
        .lines = 2,
        .pins = {
            .rs = GPIO_NUM_38,
            .e  = GPIO_NUM_37,
            .d4 = GPIO_NUM_36,
            .d5 = GPIO_NUM_35,
            .d6 = GPIO_NUM_48,
            .d7 = GPIO_NUM_47,
            .bl = HD44780_NOT_USED
        }
    };

    vTaskDelay(100 / portTICK_PERIOD_MS);  // give LCD time to power up
    hd44780_init(&lcd);
    hd44780_clear(&lcd);
  
    xTaskCreate(wiper_control_task, "Wiper_Task", 4096, NULL, 5, NULL);

    bool ignition_button_prev = false;
    bool driver_prev = false;

    printf("System Ready\n");

    while (1)
    {
        bool driver = gpio_get_level(DRI_SEAT) == 0;
        bool pass = gpio_get_level(PASS_SEAT) == 0;
        bool driver_belt = gpio_get_level(DRI_BELT) == 0;
        bool pass_belt = gpio_get_level(PASS_BELT) == 0;
        bool ignition_button = gpio_get_level(IGNI_BUTTON) == 0;

        // Welcome message
        if (driver && !driver_prev)
        {
            printf("Welcome to enhanced alarm system model 218-W26\n");
        }
        driver_prev = driver;

        bool can_start = driver && pass && driver_belt && pass_belt;

        // Green LED 
        if (can_start && !ignition)
            gpio_set_level(GREEN_LED, 1);
        else
            gpio_set_level(GREEN_LED, 0);

        // Ignition button pressed
        if (ignition_button && !ignition_button_prev)
        {
            if (can_start && !ignition)
            {
                ignition = true;
                gpio_set_level(GREEN_LED, 0);
                gpio_set_level(RED_LED, 1);
                printf("Engine started.\n");
            }
            else if (ignition)
            {
                ignition = false;
                gpio_set_level(RED_LED, 0);
            }
            else
            {
                gpio_set_level(BUZZER, 1);
                printf("Ignition inhibited\n");

                if (!driver) printf("Driver seat not occupied\n");
                if (!driver_belt) printf("Driver seatbelt not fastened\n");
                if (!pass) printf("Passenger seat not occupied\n");
                if (!pass_belt) printf("Passenger seatbelt not fastened\n");

                vTaskDelay(BUZZER_TIME / portTICK_PERIOD_MS);
                gpio_set_level(BUZZER, 0);
            }
        }
        ignition_button_prev = ignition_button;

        // Potentiometers
        int mode_raw, delay_raw;
        adc_oneshot_read(adc_handle, MODE_POT, &mode_raw);
        adc_oneshot_read(adc_handle, DELAY_POT, &delay_raw);

        // Wiper modes
        if (!ignition)
        {
            wipe_off = true;
            wipe_int = wipe_low = wipe_high = false;
        }
        else
        {
            wipe_off = mode_raw > INT_THRESH;
            wipe_int = mode_raw <= INT_THRESH && mode_raw > LOW_THRESH;
            wipe_low = mode_raw <= LOW_THRESH && mode_raw > HIGH_THRESH;
            wipe_high = mode_raw <= HIGH_THRESH;
        }

        short_del = delay_raw < SHORT_THRESH;
        med_del = delay_raw >= SHORT_THRESH && delay_raw < MED_THRESH;
        long_del = delay_raw >= MED_THRESH;

        // LCD
        if (ignition)
        {
            hd44780_clear(&lcd);
            hd44780_gotoxy(&lcd, 0, 0);

            if (wipe_off) hd44780_puts(&lcd, "OFF");
            else if (wipe_low) hd44780_puts(&lcd, "LOW");
            else if (wipe_high) hd44780_puts(&lcd, "HIGH");
            else if (wipe_int)
            {
                hd44780_puts(&lcd, "INT");
                hd44780_gotoxy(&lcd, 0, 1);
                if (short_del) hd44780_puts(&lcd, "SHORT");
                else if (med_del) hd44780_puts(&lcd, "MEDIUM");
                else hd44780_puts(&lcd, "LONG");
            }
        }

        vTaskDelay(LOOP_DELAY_MS / portTICK_PERIOD_MS);
    }
}

//wiper task for proper mode
void wiper_control_task(void *pvParameters)
{
    while (1)
    {
        if (!ignition)
        {
            perform_wipe_cycle(low_steps);   // finish cycle
            vTaskDelay(LOOP_DELAY_MS / portTICK_PERIOD_MS);
            continue;
        }

        if (wipe_high)
        {
            perform_wipe_cycle(high_steps);
        }
        else if (wipe_low)
        {
            perform_wipe_cycle(low_steps);
        }
        else if (wipe_int)
        {
            perform_wipe_cycle(low_steps);

            int delay = short_del ? short_del_ms :
                        med_del ? med_del_ms :
                                  long_del_ms;

            vTaskDelay(delay / portTICK_PERIOD_MS);
        }
        else
        {
            perform_wipe_cycle(low_steps); // finish cycle before stopping
        }
    }
}

//init functions to configure pins
static void ledc_init(void) {
// Prepare and then apply the LEDC PWM timer configuration
ledc_set_duty(LEDC_MODE, LEDC_CHANNEL, WIPERS_MIN);
ledc_update_duty(LEDC_MODE, LEDC_CHANNEL);
ledc_timer_config_t ledc_timer = {
.speed_mode = LEDC_MODE,
.duty_resolution = LEDC_DUTY_RES,
.timer_num = LEDC_TIMER,
.freq_hz = LEDC_FREQUENCY, // Set output frequency at 50 Hz
.clk_cfg = LEDC_AUTO_CLK
};
ESP_ERROR_CHECK(ledc_timer_config(&ledc_timer));

// Prepare and then apply the LEDC PWM channel configuration
ledc_channel_config_t ledc_channel = {
.speed_mode = LEDC_MODE,
.channel = LEDC_CHANNEL,
.timer_sel = LEDC_TIMER,
.intr_type = LEDC_INTR_DISABLE,
.gpio_num = LEDC_OUTPUT_IO,
.duty = 0, // Set duty to 0%
.hpoint = 0
};
ledc_channel_config(&ledc_channel);
}

//performs a wipe with the given number of steps
void perform_wipe_cycle(int steps) {

int angle = WIPERS_MIN;
int interval = (WIPERS_MAX - WIPERS_MIN) / steps;
while(angle < WIPERS_MAX) {
ledc_set_duty(LEDC_MODE, LEDC_CHANNEL, angle);
ledc_update_duty(LEDC_MODE, LEDC_CHANNEL);
angle += interval;
vTaskDelay(LOOP_DELAY_MS/portTICK_PERIOD_MS);
}
angle -= interval;
while(angle > WIPERS_MIN) {
ledc_set_duty(LEDC_MODE, LEDC_CHANNEL, angle);
ledc_update_duty(LEDC_MODE, LEDC_CHANNEL);
angle -= interval;
vTaskDelay(LOOP_DELAY_MS/portTICK_PERIOD_MS);
}
ledc_set_duty(LEDC_MODE, LEDC_CHANNEL, WIPERS_MIN);
ledc_update_duty(LEDC_MODE, LEDC_CHANNEL);
}
